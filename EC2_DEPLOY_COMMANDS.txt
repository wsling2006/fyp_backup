# Quick Deploy Commands for EC2
# Copy and paste these commands one by one into your EC2 terminal

# 1. Navigate to project
cd ~/fyp_system

# 2. Pull latest changes
git pull origin main

# 3. Rebuild backend
cd ~/fyp_system/backend && npm run build

# 4. Rebuild frontend  
cd ~/fyp_system/frontend && npm run build

# 5. Restart services
pm2 restart backend
pm2 restart frontend

# 6. Check status
pm2 status

# 7. View logs (optional)
pm2 logs backend --lines 20

# Done! Test the silent parameter feature in your browser
